    const totalExercises = Object.values(routine)

      .reduce((sum,day)=>(sum + day.exercises.length),0);


    const tabs        = document.querySelectorAll('.tab');

    const cards       = document.getElementById('cards');

    const subTitle    = document.getElementById('sub-title');

    const subTime     = document.getElementById('sub-time');

    const sep         = document.querySelector('.progress-subheader .separator');

    const clockIcon   = document.querySelector('.progress-subheader i');

    const progBar     = document.getElementById('progress-bar');

    const progCnt     = document.getElementById('progress-count');

    const landing     = document.getElementById('landing');

    const app         = document.getElementById('app');

    const startBtn    = document.getElementById('start-btn');

    const weekModal   = document.getElementById('week-modal');

    const resetWeek   = document.getElementById('reset-week');


    let initialRun = true;


    function capitalize(s){ return s.charAt(0).toUpperCase()+s.slice(1); }


    function renderDay(day){

      const dataObj = routine[day];

      subTitle.textContent = dataObj.title;

      subTime.textContent  = dataObj.time;


      // if rest day, hide only separator, clock, and time

      if (dataObj.exercises.length === 0) {

        sep.style.display       = 'none';

        clockIcon.style.display = 'none';

        subTime.style.display   = 'none';

      } else {

        sep.style.display       = '';

        clockIcon.style.display = '';

        subTime.style.display   = '';

      }


      cards.innerHTML = '';

      if (dataObj.exercises.length === 0) {

        cards.innerHTML = `<div class="rest-day">${dataObj.title}</div>`;

        updateProgress();

        return;

      }

      dataObj.exercises.forEach((ex,i)=>{

        const card = document.createElement('div');

        card.className='card';

        card.innerHTML=`

          <img src="images/${ex.img}" alt="${ex.name}">

          <div class="card-body">

            <div class="badges">

              ${ex.main?`<div class="badge-main">${capitalize(ex.main)}</div>`:''}

              ${ex.sec?`<div class="badge-secondary">${capitalize(ex.sec)}</div>`:''}

            </div>

            <div class="card-title">${capitalize(ex.name)}</div>

            ${ex.details.map(d=>`<div class="detail-line">${d}</div>`).join('')}

            <label class="form-check">

              <input type="checkbox" id="${day}-ex-${i}">

              Completed

            </label>

          </div>`;

        cards.appendChild(card);

      });

      cards.scrollTop = 0;

      cards.querySelectorAll('input[type=checkbox]').forEach(cb=>{

        cb.checked = localStorage.getItem(cb.id)==='true';

        cb.addEventListener('change', ()=>{

          localStorage.setItem(cb.id, cb.checked);

          updateProgress();

        });

      });

      updateProgress();

    }


    function updateProgress(){

      let done = 0;

      for (const day in routine) {

        routine[day].exercises.forEach((_,i)=>{

          if (localStorage.getItem(`${day}-ex-${i}`)==='true') done++;

        });

      }


      // on first load, if already complete, clear all

      if (initialRun && done === totalExercises) {

        for (const day in routine) {

          routine[day].exercises.forEach((_,i)=>{

            localStorage.removeItem(`${day}-ex-${i}`);

          });

        }

        done = 0;

      }


      progCnt.textContent = `${done}/${totalExercises}`;

      progBar.style.width = totalExercises ? (done/totalExercises*100)+'%' : '0%';


      // only trigger on new completion

      if (!initialRun && done === totalExercises) {

        confetti({spread:70,origin:{y:0.6}});

        weekModal.classList.remove('hidden');

      }


      initialRun = false;

    }


    tabs.forEach(tab=>{

      tab.addEventListener('click', ()=>{

        tabs.forEach(t=>t.classList.remove('active'));

        tab.classList.add('active');

        renderDay(tab.dataset.day);

      });

    });


    startBtn.addEventListener('click', ()=>{

      landing.style.display='none';

      app.style.display='flex';

    });


    resetWeek.addEventListener('click', ()=>{

      for (const day in routine) {

        routine[day].exercises.forEach((_,i)=>{

          localStorage.removeItem(`${day}-ex-${i}`);

        });

      }

      weekModal.classList.add('hidden');

      initialRun = true;

      // return to Monday

      tabs.forEach(t=>t.classList.remove('active'));

      const m = document.querySelector('.tab[data-day="monday"]');

      m.classList.add('active');

      renderDay('monday');

    });


    // initial render

    renderDay('monday');
